@Getter
@Setter
@Builder
@NoArgsConstructor
@AllArgsConstructor
public class VatRawData {

    private String dealNo;
    private String stepType;
    private String stepNo;
    private String chargeId;

    private Integer actualRelDate;

    private String finBatch;
    private String refStepId;

    private BigDecimal vatRate;

    private String drCalcCurr;
    private String crCalcCurr;

    private BigDecimal chargeAmt;
    private BigDecimal taxAmount;

    private BigDecimal exchangeRate;

    private String revenueType;

    private String customerId;
}

@Component
public class VatAssembler {

    public Vat map(VatRawData raw) {

        BigDecimal amountAfterVat =
                calculateAmountAfterVat(raw);

        BigDecimal convertRate =
                calculateConvertRate(raw);

        return Vat.builder()
                .VATREFID(buildVatRefId(raw))
                .VATGenDate(raw.getActualRelDate())
                .DealId(buildDealId(raw.getDealNo()))
                .DealTransDate(raw.getActualRelDate())
                .DealAccountingDate(raw.getActualRelDate())
                .REVERSE(buildReverse(raw))
                .ORIGINALVATREFID(buildOriginalVatRefId(raw))
                .VATRate(raw.getVatRate())
                .OriginalCurrency(buildCurrency(raw))
                .TotalAmount(raw.getChargeAmt())
                .TaxAmount(raw.getTaxAmount())
                .AmountAfterVAT(amountAfterVat)
                .TargetCurrency("CNY")
                .ConvertRate(convertRate)
                .ConvertedTotalAmount(
                        multiply(
                                raw.getChargeAmt(),
                                convertRate))
                .ConvertedTaxAmount(
                        multiply(
                                raw.getTaxAmount(),
                                convertRate))
                .ConvertedAmountAfterVAT(
                        multiply(
                                amountAfterVat,
                                convertRate))
                .RevenueType(raw.getRevenueType())
                .CustomerID(raw.getCustomerId())
                .build();
    }

    private String buildVatRefId(VatRawData raw) {

        return raw.getDealNo()
                + raw.getStepType()
                + raw.getStepNo()
                + raw.getChargeId();
    }

    private String buildDealId(String dealNo) {

        String location;

        String branch =
                dealNo.substring(2, 5);

        switch (branch) {

            case "860":
                location = "BEI";
                break;

            case "870":
                location = "SHA";
                break;

            case "880":
                location = "GUA";
                break;

            default:
                location = "UNKNOWN";
        }

        return dealNo.substring(2, 7)
                + dealNo.substring(7)
                + location;
    }

    private Boolean buildReverse(VatRawData raw) {

        if ("MSC".equals(raw.getStepType())
                && "AA".equals(raw.getFinBatch())) {

            return true;
        }

        return List.of(
                "802227",
                "802226",
                "80997")
                .contains(raw.getChargeId())
                && "MSC".equals(raw.getStepType())
                && List.of(
                        "RD",
                        "AM")
                        .contains(raw.getFinBatch());
    }

    private String buildOriginalVatRefId(
            VatRawData raw) {

        if ("MSC".equals(raw.getStepType())
                && List.of(
                        "AA",
                        "RD",
                        "AM")
                        .contains(raw.getFinBatch())) {

            return raw.getDealNo()
                    + raw.getRefStepId()
                    + raw.getChargeId();
        }

        return null;
    }

    private String buildCurrency(
            VatRawData raw) {

        return raw.getDrCalcCurr() != null
                && !raw.getDrCalcCurr().isBlank()
                ? raw.getDrCalcCurr()
                : raw.getCrCalcCurr();
    }

    private BigDecimal calculateAmountAfterVat(
            VatRawData raw) {

        return raw.getChargeAmt()
                .subtract(raw.getTaxAmount());
    }

    private BigDecimal calculateConvertRate(
            VatRawData raw) {

        if ("CNY".equals(buildCurrency(raw))) {
            return null;
        }

        return raw.getExchangeRate();
    }

    private BigDecimal multiply(
            BigDecimal amount,
            BigDecimal rate) {

        if (amount == null || rate == null) {
            return null;
        }

        return amount.multiply(rate);
    }
}

@Component
public class VatRawDataRowMapper
        implements RowMapper<VatRawData> {

    @Override
    public VatRawData mapRow(
            ResultSet rs,
            int rowNum)
            throws SQLException {

        return VatRawData.builder()
                .dealNo(rs.getString("DEAL_NO"))
                .stepType(rs.getString("STEP_TYPE"))
                .stepNo(rs.getString("STEP_NO"))
                .chargeId(rs.getString("CHARGE_ID"))
                .actualRelDate(
                        rs.getInt("ACTUAL_REL_DATE"))
                .finBatch(
                        rs.getString("FIN_BATCH"))
                .refStepId(
                        rs.getString("REF_STEP_ID"))
                .vatRate(
                        rs.getBigDecimal("VAT_RATE"))
                .drCalcCurr(
                        rs.getString("DR_CALC_CURR"))
                .crCalcCurr(
                        rs.getString("CR_CALC_CURR"))
                .chargeAmt(
                        rs.getBigDecimal("CHARGE_AMT"))
                .taxAmount(
                        rs.getBigDecimal("TAX_AMOUNT"))
                .exchangeRate(
                        rs.getBigDecimal("EXCHANGE_RATE"))
                .revenueType(
                        rs.getString("REVENUE_TYPE"))
                .customerId(
                        rs.getString("CUSTOMER_ID"))
                .build();
    }
}
