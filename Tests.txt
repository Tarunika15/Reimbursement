@Test
void shouldGenerateParticipantsFileSuccessfully() {

    ParticipantsRepository repository =
            mock(ParticipantsRepository.class);

    when(repository.fetch())
            .thenReturn(
                    List.of(createParticipantsRecord())
            );

    ParticipantsProcessor processor =
            new ParticipantsProcessor(repository, 9850L);

    String filePath = processor.process();

    File file = new File(filePath);

    assertTrue(file.exists());
    assertTrue(file.length() > 0);
}

private ParticipantsRecord createParticipantsRecord() {

    ParticipantsRecord record = new ParticipantsRecord();

    record.setEntity("SG");
    record.setDealNo("D001");
    record.setPartyCode("P001");
    record.setPartyId("PTY001");
    record.setCreateStepId("STEP001");
    record.setPartyExt("EXT001");
    record.setPartyName("Party Name");
    record.setPartyAddress("Party Address");
    record.setStepId("STP001");
    record.setWithinGroup("Y");
    record.setCountryCode("IND");

    return record;
}
