@Component
public class SqlFileLoader {

    public String load(String path) {

        try {

            ClassPathResource resource =
                    new ClassPathResource(path);

            return new String(
                    resource.getInputStream()
                            .readAllBytes(),
                    StandardCharsets.UTF_8);

        } catch (IOException exception) {

            throw new RuntimeException(
                    "Unable to load SQL file",
                    exception);
        }
    }
}


@Repository
@RequiredArgsConstructor
public class VatRepositoryAdaptor
        implements VatRepository {

    @Qualifier("vatDb2JdbcTemplate")
    private final NamedParameterJdbcTemplate jdbcTemplate;

    private final SqlFileLoader sqlFileLoader;

    @Value("${service.database.imex-schema}")
    private String imexSchema;

    @Value("${service.database.petra-schema}")
    private String petraSchema;

    private final VatRowMapper vatRowMapper;

    @Override
    public List<VatRawData> fetchVatData(
            final VatCommand command) {

        String sql =
                sqlFileLoader.load(
                        "sql/vat/vat-query.sql");

        sql = sql.replace(
                "${IMEX_SCHEMA}",
                imexSchema);

        sql = sql.replace(
                "${PETRA_SCHEMA}",
                petraSchema);

        Map<String, Object> params =
                new HashMap<>();

        params.put(
                "processingDateStart",
                command.getProcessingDateStart());

        params.put(
                "processingDateEnd",
                command.getProcessingDateEnd());

        System.out.println(
                "========== FINAL SQL ==========");

        System.out.println(sql);

        System.out.println(
                "================================");

        return jdbcTemplate.query(
                sql,
                params,
                vatRowMapper);
    }
}
